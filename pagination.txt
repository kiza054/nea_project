#### python function for paginating downloads table ####

def download_timetables():
    page = request.args.get('page', 1, type=int)
    timetables = Timetables.query.paginate(
        page, app.config['DOWNLOADS_PER_PAGE'], False)
    next_url = url_for('download_timetables', page=timetables.next_num) \
        if timetables.has_next else None
    prev_url = url_for('download_timetables', page=timetables.prev_num) \
        if timetables.has_prev else None
    return render_template('download_timetables.html', title='Download Timetables',
                                timetables=timetables.items, next_url=next_url, prev_url=prev_url)

#### python function for paginating bus_stops table ####

def bus_stops():
    page = request.args.get('page', 1, type=int)
    stops = Bus_stops.query.paginate(
        page, app.config['STOPS_PER_PAGE'], False)
    next_url = url_for('bus_stops', page=stops.next_num) \
        if stops.has_next else None
    prev_url = url_for('download_timetables', page=stops.prev_num) \
        if stops.has_prev else None
    return render_template('bus_stops.html', title='Bus Stop Details', stops=stops.items, next_url=next_url, prev_url=prev_url)
